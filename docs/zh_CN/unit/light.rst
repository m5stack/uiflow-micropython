    import os, sys, io
    import M5
    from M5 import *
    from unit import *


    light_0 = None


    def setup():
    global light_0

    light_0 = Light((8,9))
    M5.begin()
    Widgets.fillScreen(0x222222)

    print(light_0.get_analog_value())

Micropython Example::

    import os, sys, io
    import M5
    from M5 import *
    from hardware import *
    import time
    from unit import *

    i2c0 = None
    op90_0 = None

    def setup():
    global i2c0, op90_0

    i2c0 = I2C(0, scl=Pin(1), sda=Pin(2), freq=100000)
    op90_0 = OPUnit((8, 9), type=1)
    M5.begin()
    Widgets.fillScreen(0x222222)

    print(op90_0.get_value)
    time.sleep(1)
    print(op90_0.count_value)
    time.sleep(1)
    op90_0.count_reset()
    time.sleep(1)

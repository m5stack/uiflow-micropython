Micropython Example::

    import os, sys, io
    import M5
    from M5 import *
    from hardware import *
    from unit import *

    i2c0 = None
    dlight_0 = None

    def setup():
    global i2c0, dlight_0

    i2c0 = I2C(0, scl=Pin(1), sda=Pin(2), freq=100000)
    dlight_0 = DLight(i2c0)
    print(dlight_0.get_lux())
    M5.begin()
    Widgets.fillScreen(0x222222)

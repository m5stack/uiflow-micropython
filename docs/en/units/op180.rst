Micropython Example::

    import os, sys, io
    import M5
    from M5 import *
    import time
    from unit import *

    op180_0 = None

    def setup():
    global op180_0

    op180_0 = OPUnit((8, 9), type=1)
    M5.begin()
    Widgets.fillScreen(0x222222)

    print(op180_0.get_value)
    time.sleep(1)
    print(op180_0.count_value)
    time.sleep(1)
    op180_0.count_reset()
    time.sleep(1)
